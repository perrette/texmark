%% texmark Jinja2 template for classicthesis-style manuscripts (theses).
%%
%% Uses \documentclass{scrreprt} (KOMA-Script report) plus the bundled
%% classicthesis package for André Miede's Bringhurst-inspired typography.
%% Like the report template it has no \frontmatter / \mainmatter /
%% \backmatter macros (scrreprt lacks them); top-level embeds emit
%% \include{stem} (texmark-embed class-aware emission).
%%
%% The classicthesis package options are configurable via the
%% `classicthesis-options:` YAML key, surfaced to this template as the
%% Jinja variable `classicthesis_options` (Jinja cannot reference the
%% hyphenated key directly). Absent -> the package's own defaults.
%%
%% classicthesis.sty is bundled under this template directory and is
%% GPL-licensed; see the NOTICE file alongside it.

\documentclass[$if(documentclassoptions)$$documentclassoptions$$else$twoside,openright,titlepage,numbers=noenddot,headinclude,a4paper,11pt,BCOR=5mm,DIV=12$endif$]{scrreprt}

%% Fonts and microtypography
\usepackage[T1]{fontenc}
\usepackage[utf8]{inputenc}
\usepackage{lmodern}
\usepackage{microtype}

%% Math and figures
\usepackage{amsmath,amssymb,amsfonts,bm}
\usepackage{graphicx}
\usepackage{booktabs}
\usepackage{caption}
\usepackage{subcaption}

%% References and hyperlinks (load hyperref last among these)
\usepackage{natbib}
\usepackage{url}
\usepackage[colorlinks=true,
            linkcolor=blue,
            citecolor=blue,
            urlcolor=blue]{hyperref}

%% classicthesis typography (Bringhurst-inspired). Options come from the
%% `classicthesis-options:` YAML key when set, otherwise the package
%% defaults apply.
{% if classicthesis_options %}\usepackage[{{ classicthesis_options }}]{classicthesis}{% else %}\usepackage{classicthesis}{% endif %}

{% set already_included_packages = [
  "fontenc", "inputenc", "lmodern", "microtype",
  "amsmath", "amssymb", "amsfonts", "bm", "graphicx", "booktabs", "caption", "subcaption",
  "natbib", "url", "hyperref", "classicthesis",
] -%}
{% for package in packages -%}
{% if package not in already_included_packages -%}
\usepackage{ {{package}} }
{% endif -%}
{% endfor %}

%% Title block
\title{ {{title}} }

{% if authors -%}
{% for author in authors -%}
\author{ {{author.firstname}}~{{author.lastname}}{% if author.email %}\thanks{ \href{mailto:{{author.email}}}{ {{author.email}} } }{% endif %} }
{% endfor -%}
{% else -%}
\author{ {{ author if author else "" }} }
{% endif %}

\date{ {{ date if date else "" }} }

{{ xr_preamble | default("") }}
{{ user_preamble | default("") }}
{{ includeonly | default("") }}

\begin{document}

\maketitle

{% if dedication %}
\newpage
\thispagestyle{empty}
\vspace*{\fill}
\begin{center}
\textit{ {{dedication | join_if_list}} }
\end{center}
\vspace*{\fill}
\newpage
{% endif %}

{% if preface %}
\chapter*{Preface}
{{preface | join_if_list}}
{% endif %}

{% if foreword %}
\chapter*{Foreword}
{{foreword | join_if_list}}
{% endif %}

{% if abstract -%}
\begin{abstract}
{{ abstract | join_if_list }}
\end{abstract}
{% endif %}

\tableofcontents

{% if list_of_figures %}
\listoffigures
{% endif %}

{% if list_of_tables %}
\listoftables
{% endif %}

{{body}}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% REFERENCES
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
{% if bibliography -%}
\bibliographystyle{ {{ biblio_style if biblio_style else "plainnat" }} }
\bibliography{ {{bibliography}} }
{% endif %}

\end{document}
