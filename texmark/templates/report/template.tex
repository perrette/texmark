%% texmark Jinja2 template for report-class manuscripts (technical reports, theses).
%%
%% Uses \documentclass{report}. Unlike the book template, the report class
%% has no sectioning macros for preliminary/main/back matter; those are omitted.
%% Top-level embeds emit \include{stem} (texmark-embed class-aware emission).

\documentclass[$if(documentclassoptions)$$documentclassoptions$$else$11pt,a4paper$endif$]{report}

%% Fonts and microtypography
\usepackage[T1]{fontenc}
\usepackage[utf8]{inputenc}
\usepackage{lmodern}
\usepackage{microtype}

%% Layout
\usepackage[margin=1in]{geometry}

%% Math and figures
\usepackage{amsmath,amssymb,amsfonts,bm}
\usepackage{graphicx}
\usepackage{booktabs}
\usepackage{caption}
\usepackage{subcaption}

%% References and hyperlinks (load hyperref last among these)
\usepackage{natbib}
\usepackage{url}
\usepackage[colorlinks=true,
            linkcolor=blue,
            citecolor=blue,
            urlcolor=blue]{hyperref}

{% set already_included_packages = [
  "fontenc", "inputenc", "lmodern", "microtype", "geometry",
  "amsmath", "amssymb", "amsfonts", "bm", "graphicx", "booktabs", "caption", "subcaption",
  "natbib", "url", "hyperref",
] -%}
{% for package in packages -%}
{% if package not in already_included_packages -%}
\usepackage{ {{package}} }
{% endif -%}
{% endfor %}

%% Title block
\title{ {{title}} }

{% if authors -%}
{% for author in authors -%}
\author{ {{author.firstname}}~{{author.lastname}}{% if author.email %}\thanks{ \href{mailto:{{author.email}}}{ {{author.email}} } }{% endif %} }
{% endfor -%}
{% else -%}
\author{ {{ author if author else "" }} }
{% endif %}

\date{ {{ date if date else "" }} }

{{ xr_preamble | default("") }}
{{ user_preamble | default("") }}
{{ includeonly | default("") }}

\begin{document}

\maketitle

{% if dedication %}
\newpage
\thispagestyle{empty}
\vspace*{\fill}
\begin{center}
\textit{ {{dedication | join_if_list}} }
\end{center}
\vspace*{\fill}
\newpage
{% endif %}

{% if preface %}
\chapter*{Preface}
{{preface | join_if_list}}
{% endif %}

{% if foreword %}
\chapter*{Foreword}
{{foreword | join_if_list}}
{% endif %}

{% if abstract -%}
\begin{abstract}
{{ abstract | join_if_list }}
\end{abstract}
{% endif %}

\tableofcontents

{% if list_of_figures %}
\listoffigures
{% endif %}

{% if list_of_tables %}
\listoftables
{% endif %}

{{body}}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% REFERENCES
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
{% if bibliography -%}
\bibliographystyle{ {{ biblio_style if biblio_style else "plainnat" }} }
\bibliography{ {{bibliography}} }
{% endif %}

\end{document}
